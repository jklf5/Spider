如今市面上关于动画的开源库多得数不胜数，有关于CSS、js甚至是canvas渲染的，百花齐放，效果炫酷。但你是否曾想过，自己亲手去实现（封装）一个简单的动画库？

本文将从零开始，讲授如何搭建一个简单的动画库，它将具备以下几个特征：

  • 从实际动画中抽象出来，根据给定的动画速度曲线，完成“由帧到值”的计算过程，而实际渲染则交给开发者决定，更具拓展性

  • 支持基本的事件监听，如 onPlay、 onStop、 onReset、 onEnd，及相应的回调函数

  • 支持手动式触发动画的各种状态，如 play、 stop、 reset、 end

  • 支持自定义路径动画

  • 支持多组动画的链式触发

/timeline 是本项目的根目录，各文件的作用分别如下：

引入缓动函数

所谓动画，简单来说，就是在一段时间内不断改变目标某些状态的结果。这些状态值在运动过程中，随着时间不断发生变化，状态值与时间存在一一对应的关系，这就是所谓的“帧-值”对应关系，常说的动画缓动函数也是相同的道理。

有了这种函数关系，给定任意一个时间点，我们都能计算出对应的状态值。OK，那如何在动画中引入缓动函数呢？不说废话，直接上代码。

首先我们在core.js中创建了一个 Core类：

  1. class Core {
  2.   constructor(opt) {
  3.     // 初始化，并将实例当前状态设置为'init'
  4.     this._init(opt);
  5.     this.state = 'init';
  6.   }
  8.   _init(opt) {
  9.     this._initValue(opt.value);
  10.     // 保存动画总时长、缓动函数以及渲染函数
  11.     this.duration = opt.duration || 1000;
  12.     this.timingFunction = opt.timingFunction || 'linear';
  13.     this.renderFunction = opt.render || this._defaultFunc;
  15.     // 未来会用到的事件函数
  16.     this.onPlay = opt.onPlay;
  17.     this.onEnd = opt.onEnd;
  18.     this.onStop = opt.onStop;
  19.     this.onReset = opt.onReset;
  20.   }
  22.   _initValue(value) {
  23.     // 初始化运动值
  24.       this.value = [];
  25.       value.forEach(item => {
  26.         this.value.push({
  27.           start: parseFloat(item[0]),
  28.           end: parseFloat(item[1]),
  29.         });
  30.       });
  31.   }
  32. }

我们在构造函数中对实例调用 _init函数，对其初始化：将传入的参数保存在实例属性中。

当你看到 _initValue的时候可能不大明白：外界传入的 value到底是啥？其实 value是一个数组，它的每一个元素都保存着独立动画的起始与结束两种状态。这样说好像有点乱，举个栗子好了：假设我们要创建一个动画，让页面上的div同时往右、左分别平移300px、500px，此外还同时把自己放大1.5倍。在这个看似复杂的动画过程中，其实可以拆解成三个独立的动画，每一动画都有自己的起始与终止值：

  • 对于往右平移，就是把css属性的 translateX的0px变成了300px

  • 同理，往下平移，就是把 tranlateY的0px变成500px

  • 放大1.5倍，也就是把 `scale从1变成1.5

因此传入的value应该长成这样： [[0,300],[0,500],[1,1.5]] 。我们将数组的每一个元素依次保存在实例的value属性中。

此外， renderFunction是由外界提供的渲染函数，即 opt.render，它的作用是：动画运动的每一帧，都会调用一次该函数，并把计算好的当前状态值以参数形式传入，有了当前状态值，我们就可以自由地选择渲染动画的方式啦。

接下来我们给Core类添加一个循环函数：

  1. _loop() {
  2.   const t = Date.now() - this.beginTime,
  3.         d = this.duration,
  4.         func = Tween[this.timingFunction] || Tween['linear'];
  6.   if (t >= d) {
  7.     this.state = 'end';
  8.     this._renderFunction(d, d, func);
  9.   } else {
  10.     this._renderFunction(t, d, func);
  11.     window.requestAnimationFrame(this._loop.bind(this));
  12.   }
  13. }
  15. _renderFunction(t, d, func) {
  16.   const values = this.value.map(value => func(t, value.start, value.end - value.start, d));
  17.   this.renderFunction.apply(this, values);
  18. }

_loop的作用是：倘若当前时间进度 t还未到终点，则根据当前时间进度计算出目标现在的状态值，并以参数的形式传给即将调用的渲染函数，即 renderFunction，并继续循环。如果大于 duration，则将目标的运动终止值传给 renderFunction，运动结束，将状态设为 end。

代码中的 Tween是从tween.js文件引入的缓动函数，tween.js的代码如下（网上搜搜基本都差不多= =）:

  1. /*
  2.  * t: current time（当前时间）；
  3.  * b: beginning value（初始值）；
  4.  * c: change in value（变化量）；
  5.  * d: duration（持续时间）。
  6.  * Get effect on 'http://easings.net/zh-cn'
  7.  */
  9. const Tween = {
  10.     linear: function (t, b, c, d) {
  11.         return c * t / d + b;
  12.     },
  13.     // Quad
  14.     easeIn: function (t, b, c, d) {
  15.         return c * (t /= d) * t + b;
  16.     },
  17.     easeOut: function (t, b, c, d) {
  18.         return -c * (t /= d) * (t - 2) + b;
  19.     },
  20.     easeInOut: function (t, b, c, d) {
  21.         if ((t /= d / 2) < 1) return c / 2 * t * t + b;
  22.         return -c / 2 * ((--t) * (t - 2) - 1) + b;
  23.     },
  24.     // Cubic
  25.     easeInCubic: function (t, b, c, d) {
  26.         return c * (t /= d) * t * t + b;
  27.     },
  28.     easeOutCubic: function (t, b, c, d) {
  29.         return c * ((t = t / d - 1) * t * t + 1) + b;
  30.     },
  31.     easeInOutCubic: function (t, b, c, d) {
  32.         if ((t /= d / 2) < 1) return c / 2 * t * t * t + b;
  33.         return c / 2 * ((t -= 2) * t * t + 2) + b;
  34.     },
  35.     // Quart
  36.     easeInQuart: function (t, b, c, d) {
  37.         return c * (t /= d) * t * t * t + b;
  38.     },
  39.     easeOutQuart: function (t, b, c, d) {
  40.         return -c * ((t = t / d - 1) * t * t * t - 1) + b;
  41.     },
  42.     easeInOutQuart: function (t, b, c, d) {
  43.         if ((t /= d / 2) < 1) return c / 2 * t * t * t * t + b;
  44.         return -c / 2 * ((t -= 2) * t * t * t - 2) + b;
  45.     },
  46.     // Quint
  47.     easeInQuint: function (t, b, c, d) {
  48.         return c * (t /= d) * t * t * t * t + b;
  49.     },
  50.     easeOutQuint: function (t, b, c, d) {
  51.         return c * ((t = t / d - 1) * t * t * t * t + 1) + b;
  52.     },
  53.     easeInOutQuint: function (t, b, c, d) {
  54.         if ((t /= d / 2) < 1) return c / 2 * t * t * t * t * t + b;
  55.         return c / 2 * ((t -= 2) * t * t * t * t + 2) + b;
  56.     },
  57.     // Sine
  58.     easeInSine: function (t, b, c, d) {
  59.         return -c * Math.cos(t / d * (Math.PI / 2)) + c + b;
  60.     },
  61.     easeOutSine: function (t, b, c, d) {
  62.         return c * Math.sin(t / d * (Math.PI / 2)) + b;
  63.     },
  64.     easeInOutSine: function (t, b, c, d) {
  65.         return -c / 2 * (Math.cos(Math.PI * t / d) - 1) + b;
  66.     },
  67.     // Expo
  68.     easeInExpo: function (t, b, c, d) {
  69.         return (t == 0) ? b : c * Math.pow(2, 10 * (t / d - 1)) + b;
  70.     },
  71.     easeOutExpo: function (t, b, c, d) {
  72.         return (t == d) ? b + c : c * (-Math.pow(2, -10 * t / d) + 1) + b;
  73.     },
  74.     easeInOutExpo: function (t, b, c, d) {
  75.         if (t == 0) return b;
  76.         if (t == d) return b + c;
  77.         if ((t /= d / 2) < 1) return c / 2 * Math.pow(2, 10 * (t - 1)) + b;
  78.         return c / 2 * (-Math.pow(2, -10 * --t) + 2) + b;
  79.     },
  80.     // Circ
  81.     easeInCirc: function (t, b, c, d) {
  82.         return -c * (Math.sqrt(1 - (t /= d) * t) - 1) + b;
  83.     },
  84.     easeOutCirc: function (t, b, c, d) {
  85.         return c * Math.sqrt(1 - (t = t / d - 1) * t) + b;
  86.     },
  87.     easeInOutCirc: function (t, b, c, d) {
  88.         if ((t /= d / 2) < 1) return -c / 2 * (Math.sqrt(1 - t * t) - 1) + b;
  89.         return c / 2 * (Math.sqrt(1 - (t -= 2) * t) + 1) + b;
  90.     },
  91.     // Elastic
  92.     easeInElastic: function (t, b, c, d, a, p) {
  93.         let s;
  94.         if (t == 0) return b;
  95.         if ((t /= d) == 1) return b + c;
  96.         if (typeof p == "undefined") p = d * .3;
  97.         if (!a || a < Math.abs(c)) {
  98.             s = p / 4;
  99.             a = c;
  100.         } else {
  101.             s = p / (2 * Math.PI) * Math.asin(c / a);
  102.         }
  103.         return -(a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
  104.     },
  105.     easeOutElastic: function (t, b, c, d, a, p) {
  106.         let s;
  107.         if (t == 0) return b;
  108.         if ((t /= d) == 1) return b + c;
  109.         if (typeof p == "undefined") p = d * .3;
  110.         if (!a || a < Math.abs(c)) {
  111.             a = c;
  112.             s = p / 4;
  113.         } else {
  114.             s = p / (2 * Math.PI) * Math.asin(c / a);
  115.         }
  116.         return (a * Math.pow(2, -10 * t) * Math.sin((t * d - s) * (2 * Math.PI) / p) + c + b);
  117.     },
  118.     easeInOutElastic: function (t, b, c, d, a, p) {
  119.         let s;
  120.         if (t == 0) return b;
  121.         if ((t /= d / 2) == 2) return b + c;
  122.         if (typeof p == "undefined") p = d * (.3 * 1.5);
  123.         if (!a || a < Math.abs(c)) {
  124.             a = c;
  125.             s = p / 4;
  126.         } else {
  127.             s = p / (2 * Math.PI) * Math.asin(c / a);
  128.         }
  129.         if (t < 1) return -.5 * (a * Math.pow(2, 10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p)) + b;
  130.         return a * Math.pow(2, -10 * (t -= 1)) * Math.sin((t * d - s) * (2 * Math.PI) / p) * .5 + c + b;
  131.     },
  132.     // Back
  133.     easeInBack: function (t, b, c, d, s) {
  134.         if (typeof s == "undefined") s = 1.70158;
  135.         return c * (t /= d) * t * ((s + 1) * t - s) + b;
  136.     },
  137.     easeOutBack: function (t, b, c, d, s) {
  138.         if (typeof s == "undefined") s = 1.70158;
  139.         return c * ((t = t / d - 1) * t * ((s + 1) * t + s) + 1) + b;
  140.     },
  141.     easeInOutBack: function (t, b, c, d, s) {
  142.         if (typeof s == "undefined") s = 1.70158;
  143.         if ((t /= d / 2) < 1) return c / 2 * (t * t * (((s *= (1.525)) + 1) * t - s)) + b;
  144.         return c / 2 * ((t -= 2) * t * (((s *= (1.525)) + 1) * t + s) + 2) + b;
  145.     },
  146.     // Bounce
  147.     easeInBounce: function (t, b, c, d) {
  148.         return c - Tween.easeOutBounce(d - t, 0, c, d) + b;
  149.     },
  150.     easeOutBounce: function (t, b, c, d) {
  151.         if ((t /= d) < (1 / 2.75)) {
  152.             return c * (7.5625 * t * t) + b;
  153.         } else if (t < (2 / 2.75)) {
  154.             return c * (7.5625 * (t -= (1.5 / 2.75)) * t + .75) + b;
  155.         } else if (t < (2.5 / 2.75)) {
  156.             return c * (7.5625 * (t -= (2.25 / 2.75)) * t + .9375) + b;
  157.         } else {
  158.             return c * (7.5625 * (t -= (2.625 / 2.75)) * t + .984375) + b;
  159.         }
  160.     },
  161.     easeInOutBounce: function (t, b, c, d) {
  162.         if (t < d / 2) {
  163.             return Tween.easeInBounce(t * 2, 0, c, d) * .5 + b;
  164.         } else {
  165.             return Tween.easeOutBounce(t * 2 - d, 0, c, d) * .5 + c * .5 + b;
  166.         }
  167.     }
  168. };
  170. export default Tween;
